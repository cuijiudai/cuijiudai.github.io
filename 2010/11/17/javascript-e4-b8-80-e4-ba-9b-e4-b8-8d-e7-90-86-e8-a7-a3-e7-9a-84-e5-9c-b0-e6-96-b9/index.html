JavaScript 一些不理解的地方

2010-11-17

Posted: 16 May 2010 07:53 PM PDT

看了一下Ruby’s Louvre的一些javascript题目有些不解的问题，记录一下


不解之一

在+、 -、 *、 /、 = 、== 、!= 、 <、 > 在这种情况下JavaScript停止了行末自动添加分号

[code=”js”]
function test() {
return 1
+2
}
alert(test())
[/code]
[code=”js”]
function test() {
return 1+
2
}
alert(test())
[/code]
[code=”js”]
//不报错
function test() {
return 1 ;+
2
}
alert(test())
[/code]

[] 的作用

[code=”js”]
alert({a: 1, b: 2}[[“b”]]) //弹出 2
alert({a: 1, b: 2}[‘b’]) //弹出 2
[/code]

第一个 两对[] 作用？

[code=”js”]
alert(typeof(‘foo’.split(‘’) + []));
[/code]

+ 的作用

[code=”js”]
alert(‘foo’.split(‘’) + []) //弹出 f,o,o
alert(‘foo’.split(‘’) + [‘t’]) // f,o,ot
[/code]

数组 + 数组 变成string

原文解释：

//参见ECMA3-5th的第9.3节与15.7.1节。如果Number([value])的value不存在，则返回0

//一个字符串被执行了“+”，“-”等操作，相当于调用了Number(str)，这时亦即相当于调用了toNumber函数

//这是一个内建函数，只存在于JS引擎中，对于这个str它会执行trim操作，再执行数字转换，

//如果能转换基本上转换为十进制(如0x12转为18)，如果数字过大会使用指数，

//如果trim后得到的为Infinity，则直接转换它，如：

var a = +' Infinity  ';

alert(typeof a)   //number

//如果不能转换则返回NaN

[code=”js”]
alert(Array(2).join()) //弹出 ,
alert(Array(3).join()) //弹出 ,,
[/code]

和+相同的原理

最不解的 .. 的作用

[code=”js”]
alert([1,2,3,4,5][0..toString.length])
[/code]

原文解释：

//0..相当于0.0.,这里不能直接写0.toString，

//因为如果一个点会提示错误 ，JS引擎无法判断是小数点还是”.”操作符

//http://www.cnblogs.com/rubylouvre/archive/2009/11/23/1608601.html

//接着就是取数字的toString方法的length属性，注意这里并没有执行toString

//所有函数都一个length属性，为其形参的个数，如

var fn = function(a,b){};

fn.length; // 2

//toString函数的形参为1，自己翻书吧，因此相当于[1,2,3,4,5][1]

[code=”js”]
var d;
Number.prototype.a =
Number.prototype.b =
Number.prototype.c =
Number.prototype;
function f1() { return 1..a.b.c }
//alert(a) //a is not defined
//alert(e) //e is not defined

alert(1..a) //弹出 0
alert(1..e) //弹出 undefined
//alert(d) //弹出 undefined
[/code]

a 和e 的区别在哪里？

原文解释：

//先看Number.prototype.a = Number.prototype;

//Number.prototype为Number的一个实例

//相当于 new Number 得0

//相当于1..a为0

//接着是0..b

//而b也是Number.prototype.b = Number.prototype;

//因此0..b为0

//最后的0..c为一样，为0
