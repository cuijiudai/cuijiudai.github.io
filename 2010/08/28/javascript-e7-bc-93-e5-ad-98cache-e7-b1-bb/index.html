JavaScript 缓存(Cache)类

2010-08-28

Posted: 08 Jul 2010 07:46 AM PDT

每一个开发人员都知道缓存的重要性，在后台我们有缓存（memcached, xcache,等）来防止数据库被刷爆，在CDN（content delivery networks）中我们有edge caching 希望网页内容被浏览多余一次时，浏览器可以缓存网页内容。这次我们给出了一个JavaScript缓存解决方案，他可以支持 HTML5的Local Storage，中文Local Storage教程可以参考Web Storage全解析。

构造的思路

[code=’js’]
function CacheProvider() {
// values will be stored here
this._cache = {};
}
[/code]

检测是否支持Local Storage

[code=’js’]
try {
CacheProvider.hasLocalStorage = (‘localStorage’ in window) &amp;&amp; window[‘localStorage’] !== null;
} catch (ex) {
CacheProvider.hasLocalStorage = false;
}
[/code]

我们使用try/catch因为虽然Firefox支持它，但是在about:config设置中默认是禁止的并且会抛出错误。

然后我们要添加local storage对存储对象的支持，这个可以从 Christopher Blizzard 的嘉文Saving data with local storage ，因为你仅可以在local storage中存储字符串。

JSON 解析

[code=’js’]
if (CacheProvider.hasLocalStorage) {
Storage.prototype.setObject = function(key, value) {
this.setItem(key, JSON.stringify(value));
};

Storage.prototype.getObject = function(key) {
return JSON.parse(this.getItem(key));
};
}
[/code]

使用了native JSON，有了核心方法，我们然后会有get，set和clear……

功能函数核心类

[code=’js’]
CacheProvider.prototype = {

/**

* {String} k - the key
* {Boolean} local - get this from local storage?
* {Boolean} o - is the value you put in local storage an object?
*/

get: function(k, local, o) {
if (local &amp;&amp; CacheProvider.hasLocalStorage) {
var action = o ? ‘getObject’ : ‘getItem’;
return localStorageaction || undefined;
} else {
return this._cache[k] || undefined;
}
},

/**

* {String} k - the key
* {Object} v - any kind of value you want to store
* however only objects and strings are allowed in local storage
* {Boolean} local - put this in local storage
*/

set: function(k, v, local) {
if (local &amp;&amp; CacheProvider.hasLocalStorage) {
if (typeof v !== ‘string’)) {
// make assumption if it’s not a string, then we’re storing an object
localStorage.setObject(k, v);
} else {
try {
localStorage.setItem(k, v);
} catch (ex) {
if (ex.name == ‘QUOTA_EXCEEDED_ERR’) {
// developer needs to figure out what to start invalidating
throw new Exception(v);
return;
}
}
}
} else {
// put in our local object
this._cache[k] = v;
}
// return our newly cached item
return v;
},

/**

* {String} k - the key
* {Boolean} local - put this in local storage
* {Boolean} o - is this an object you want to put in local storage?
*/

clear: function(k, local, o) {
if (local &amp;&amp; CacheProvider.hasLocalStorage) {
localStorage.removeItem(k);
}
// delete in both caches - doesn’t hurt.
delete this._cache[k];
}

};

var i18nCache = new CacheProvider;

if (i18nCache.get(‘topnav’)) {
$(‘#nav’).html(i18nCache.get(‘topnav’));
} else {
ajax(‘top-nav.tmpl’, function(html) {
i18nCache.set(‘topnav’, html);
$(‘#nav’).html(i18nCache.get(‘topnav’));
});
}
[/code]

如何使用这个类那

在getElementsByClassName方法中的使用

[code=’js’]
var cache = new CacheProvider;

window.getElementsByClassName = getElementsByClassName || function(c) {
var reg = cache.get(c) || cache.set(c, new RegExp(“(?:^|\s+)” + c + “(?:\s+|$)”));
var elements = document.getElementsByTagName(‘*’);
var results = [];
for (var i = 0; i < elements.length; i++) {
if (elements[i].className.match(reg)) {
results.push(elements[i]);
}
}
return results;
};
[/code]

当下次我们使用同一个类时，它会使用一个预编译的正则表达式，而不是重新构建一个新的。

下一个例子，对于需要i18n（国际化）的大型应用，我们可以缓存编译的html字符串。

[code=’js’]
var i18nCache = new CacheProvider;

if (i18nCache.get(‘topnav’)) {
$(‘#nav’).html(i18nCache.get(‘topnav’));
} else {
ajax(‘top-nav.tmpl’, function(html) {
i18nCache.set(‘topnav’, html);
$(‘#nav’).html(i18nCache.get(‘topnav’));
});
}
[/code]
