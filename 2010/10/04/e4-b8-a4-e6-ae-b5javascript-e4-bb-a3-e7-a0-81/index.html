2010-10-04

以下两段代码的区别在哪？
以下代码片段出自lifesinger的一道小题目

[code=’js’]
var o = {
m: function() {
return this.length;
},
length: 1
};
var m = o.m;
alert(m()); // 请问输出什么? 并解释为什么。
[/code]

another snippet
[code=’js’]
function o(){
var length = 5;
return function(){
alert(length);
};

}
var length = 3;
var o=o();
o();
[/code]
