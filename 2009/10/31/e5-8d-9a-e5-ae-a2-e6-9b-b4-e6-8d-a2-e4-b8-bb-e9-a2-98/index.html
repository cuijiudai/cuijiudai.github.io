博客更换主题

2009-10-31

博客改版了，以前的博客主题是出自一个外国的网站，随便翻译了一下就拿来用了，现在的这款是出自Prower,谢谢博主的分享。我对主题作了一下更改，添加了一些js，主要是图片方面，包括图片自动添加背景，圆角，最大尺寸判定,缩略图的鼠标指向放大。在此列一下代码，大家多提意见。下次有机会我会添加更多的js应用。

js代码

[code=”javascript”]

$(document).ready(function(){
//图片的背景边框效果

var offsetX = 20;
var offsetY = 10;

$(".content img").each(function(){
var contentwidth=$(".content").width();
var maxwidth=contentwidth-18; //减去padding和border的宽度
    $(this).wrap("<div></div>");
    $(this).parent().attr("class","addcaption");
    $(this).attr("class","");
    $(this).parent().append("<span>"+$(this).attr("title")+"</span>");
    thiswidth=$(this).width();
    if(maxwidth<thiswidth)
    {
        thiswidth=maxwidth;   // 图片宽度等于最大宽度
        $(this).width(thiswidth);

        //鼠标悬停的放大效果
        if(true)
        {
            var $imgsrc=$(this).attr("src");
            $(this).hover(function(e){
                $('![big image]()')
                .css('top',e.pageY +offsetY)
                .css('left', e.pageX + offsetX)
                .appendTo('body');

            }, function() {
                $("#largeImage").remove();
            });

            $(this).mousemove(function(e) {
                $("#largeImage").css('top', e.pageY + offsetY).css('left', e.pageX + offsetX);
            });
        }
    }
    $(this).parent().width(thiswidth);
    $(this).parent().corner("5px");
});

});

[/code]
